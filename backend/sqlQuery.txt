create database studentData;
use studentData;
create table studentData.studentEntryData(id int auto_increment primary key, first_name varchar(50) not null, last_name varchar(50) not null, mobile_no varchar(13), city varchar(50), email_id varchar(50) unique, dob date, created_at timestamp default current_timestamp, updated_at timestamp default current_timestamp on update current_timestamp);
create table studentData.studentMarksData(id int auto_increment primary key, student_id int not null unique, math int default 0, science int default 0, gk int default 0, marathi int default 0, english int default 0, hindi int default 0, created_at timestamp default current_timestamp, updated_at timestamp default current_timestamp on update current_timestamp, constraint fk_student foreign key (student_id) references studentEntryData(id));
